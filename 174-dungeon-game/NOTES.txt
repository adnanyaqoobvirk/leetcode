class Solution:
def calculateMinimumHP(self, dungeon: List[List[int]]) -> int:
def recurse(i: int, j: int, low: int, total: int) -> int:
if i > n or j > m:
return -2001

total = dungeon[i][j] + total
low = min(low, total)

if i == n and j == m:
return low

return max(
recurse(i + 1, j, low, total),
recurse(i, j + 1, low, total)
)

n, m = len(dungeon) - 1, len(dungeon[0]) - 1
return abs(recurse(0, 0, 0, 0)) + 1
